\documentclass{standalone}

\usepackage{tikz}
\usetikzlibrary{arrows}
\usetikzlibrary{decorations.markings}
\usetikzlibrary{calc}

\begin{document}

\definecolor{darkgreen}{RGB}{24,145,0}

\begin{tikzpicture}
    \tikzstyle{state}=[minimum width=1.5cm, font=\boldmath];
    % First row
    \node (0) at (0,0) [state] {$(0)$};
    \node (1) at ($(0)+(3,0)$) [state] {$(1)$};
    \node (2) at ($(1)+(3,0)$) [state] {$(2)$};
    \node (dots1) at ($(2)+(3,0)$) [state] {\LARGE...};

    % Second row
    \node (dots2) at ($(0)+(0,-3)$) [state] {\LARGE...};
    \node (n-1) at ($(1)+(0,-3)$) [state] {$(n-1)$};
    \node (n) at ($(n-1)+(3,0)$) [state] {$(n)$};
    \node (n+1) at ($(n)+(3,0)$) [state] {$(n+1)$};
    \node (-1) at ($(n+1)+(3,0)$) [state] {$(-1)$};

    % Transitions
    % Arrivals

    \draw[red] (0) edge[out=45,in=135,->,thick] (1);
    \draw[red] (1) edge[out=45,in=135,->,thick] (2);
    \draw[red] (2) edge[out=45,in=180,->,thick] ($(2)+(1.5,0.7)$);
    \draw[red] ($(n-1)+(-1.5,0.7)$) edge[out=0,in=135,->,thick] (n-1);
    \draw[red] (n-1) edge[out=45,in=135,->,thick] (n);
    \draw[red] (n) edge[out=45,in=135,->,thick] (n+1);

    % Services

    \draw[blue] (0) edge[out=-45,in=-135,<-,thick] (1);
    \draw[blue] (1) edge[out=-45,in=-135,<-,thick] (2);
    \draw[blue] (2) edge[out=-45,in=180,<-,thick] ($(2)+(1.5,-0.7)$);
    \draw[blue] ($(n-1)+(-1.5,-0.7)$) edge[out=0,in=-135,<-,thick] (n-1);
    \draw[blue] (n-1) edge[out=-45,in=-135,<-,thick] (n);
    \draw[blue] (n) edge[out=-45,in=-135,<-,thick] (n+1);

    % To deadlock

    \draw[darkgreen] (n+1) edge[->,thick] (-1);

\end{tikzpicture}

\end{document}
