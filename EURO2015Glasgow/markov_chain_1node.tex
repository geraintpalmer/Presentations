\documentclass{standalone}

\usepackage{tikz}
\usetikzlibrary{arrows}
\usetikzlibrary{decorations.markings}
\usetikzlibrary{calc}

\begin{document}

\definecolor{darkgreen}{RGB}{24,145,0}

\begin{tikzpicture}
    \tikzstyle{state}=[minimum width=1.5cm, font=\boldmath];
    % First row
    \node (0) at (0,0) [state] {$(0)$};
    \node (1) at ($(0)+(3,0)$) [state] {$(1)$};
    \node (2) at ($(1)+(3,0)$) [state] {$(2)$};
    \node (3) at ($(2)+(3,0)$) [state] {$(3)$};
    \node (4) at ($(3)+(3,0)$) [state] {$(4)$};
    \node (-1) at ($(4)+(3,0)$) [state] {$(-1)$};

    % Transitions
    % Arrivals

    \draw[red] (0) edge[out=45,in=135,->,ultra thick] (1);
    \draw[red] (1) edge[out=45,in=135,->,ultra thick] (2);
    \draw[red] (2) edge[out=45,in=135,->,ultra thick] (3);
    \draw[red] (3) edge[out=45,in=135,->,ultra thick] (4);

    % Services

    \draw[blue] (0) edge[out=-45,in=-135,<-,ultra thick] (1);
    \draw[blue] (1) edge[out=-45,in=-135,<-,ultra thick] (2);
    \draw[blue] (2) edge[out=-45,in=-135,<-,ultra thick] (3);
    \draw[blue] (3) edge[out=-45,in=-135,<-,ultra thick] (4);
    % To deadlock

    \draw[darkgreen] (4) edge[->,ultra thick] (-1);

\end{tikzpicture}

\end{document}
